Public Function מקור_שורה_לפי_דגם(קוד_שמלה As Integer, תאריך_לועזי, מידה As String, כמות As Integer, NewRecord As Boolean, Optional ץאריך_תפוסה_אחרון As Date) As String
      
     מקור_שורה_לפי_דגם = שמלות_מקור_שורה_לפי(1, קוד_שמלה, תאריך_לועזי, מידה, כמות, NewRecord, , , , תאריך_תפוסה_אחרון)
     
End Function

Public Function מקור_שורה_לפי_דגם_בנית_שדה(תא As ComboBox, קוד_שמלה, תאריך_לועזי, מידה As String, כמות As Integer, NewRecord As Boolean, Optional ימים_להגבלה As Variant = -1, Optional חישוב_ימים As String, Optional תאריך_תפוסה_אחרון As Date, Optional קוד_הזמנה As Long, Optional חפש_באב As Boolean = -1) As String
    
    ימים_להגבלה = MyCint(ימים_להגבלה, -1)
    Dim מתאריך As Variant, עד_תאריך As Variant
    
    Dim טופס As Object
    Dim db As DAO.Database
    Dim rs As DAO.Recordset
    Dim strSQL As String
    
    ' בדיקה האם לחפש בטופס האב (-1 שווה True ב-VBA)
    If חפש_באב = -1 Then
        
        ' --- מסלול 1: שליפת הנתונים מתוך טופס האב ---
        Set טופס = תא.Parent
        With טופס.Parent
            מתאריך = IIf(!אירוע_חול, !מתאריך, Null)
            עד_תאריך = IIf(!אירוע_חול, !עד_תאריך, Null)
        End With
        
    Else
        
        ' --- מסלול 2: שליפת הנתונים מטבלת הזמנות במסד הנתונים ---
        ' בניית השאילתה - נדרשים גם התאריכים וגם שדה אירוע_חול לטובת התנאי
        strSQL = "SELECT אירוע_חול, מתאריך, עד_תאריך FROM הזמנות WHERE קוד_הזמנה = " & קוד_הזמנה
        
        Set db = CurrentDb()
        Set rs = db.OpenRecordset(strSQL, dbOpenSnapshot)
        
        ' וידוא שההזמנה אכן קיימת בטבלה
        If Not rs.EOF Then
            ' שימוש באותה לוגיקת תנאי, אך הפעם הנתונים מגיעים ממערך הרשומות (rs)
            מתאריך = IIf(rs!אירוע_חול, rs!מתאריך, Null)
            עד_תאריך = IIf(rs!אירוע_חול, rs!עד_תאריך, Null)
        Else
            ' אם הרשומה לא נמצאה, נציב Null כדי למנוע שגיאות בהמשך
            מתאריך = Null
            עד_תאריך = Null
        End If
        
        ' סגירת מערך הרשומות ושחרור הזיכרון
        If Not rs Is Nothing Then
            rs.Close
            Set rs = Nothing
        End If
        Set db = Nothing
        
    End If
    
   ' תא.RowSource = שמלות_מקור_שורה_לפי(1, קוד_שמלה, תאריך_לועזי, מידה, כמות, NewRecord, , ימים_להגבלה, חישוב_ימים, תאריך_תפוסה_אחרון)
    Call שמלות_מלאי_חדש(תא, CDate(תאריך_לועזי), CLng(ימים_להגבלה), קוד_שמלה, , , "מידה, כמות" & IIf(ימים_להגבלה > -1, ",available_gap_" & ימים_להגבלה & " AS [ימים " & ימים_להגבלה & "]", ""), מתאריך, עד_תאריך)
  
    If ימים_להגבלה > -1 Then
      תא.ColumnCount = 3
      תא.ListWidth = 4500
      תא.ColumnWidths = "1500;1500;1500;1500"
    Else
      תא.ColumnCount = 2
      תא.ListWidth = 3000
      תא.ColumnWidths = "1500;1500;1500"
    End If

End Function

Public Function שמלות_מקור_שורה_לפי(סוג As Integer, Optional ByVal קוד_שמלה As Long, Optional תאריך_לועזי, Optional מידה As String, _
                                    Optional כמות As Integer, Optional NewRecord As Boolean, Optional סנן_לשמלה As Boolean, _
                                    Optional ByVal ימים_להגבלה As Integer = -1, Optional ByVal חישוב_ימים As String, Optional ByVal תאריך_תפוסה_אחרון As Date, _
                                    Optional ByVal מידה_קבועה As String, Optional ByVal לא_פעילים As Boolean, Optional ByVal קוד_הזמנה As Boolean) As String
   
   ' בדיקת אורך הפעולה
   Dim התחלה As Single, סוף As Single, SQL3 As String
   התחלה = Timer
     
   Dim שדה_כמות As String
   Dim תאריך_ראשון As Date, תאריך_אחרון As Date, ימים As Integer
   Dim שדה_כמות_2 As String
   Dim תאריך_ראשון_2 As Date, תאריך_אחרון_2 As Date, ימים_2 As Integer, אין_תאריך As Boolean, מסנן As String
   Dim sql As String, SQL2 As String, שמלות_במלאי_לפי_מידה
   
   ' תאריך ברירת מחדל
   If Not IsDate(תאריך_לועזי) Then תאריך_לועזי = Date: אין_תאריך = -1
      

   ימים = CInt(מהגדרות(6)) '+ 1
   ימים = יומן_סינון_תאריכים(תאריך_לועזי, ימים, False)
   תאריך_ראשון = תאריך_לשאילתה(CDate(תאריך_לועזי) - ימים)
   
   If ימים_להגבלה > -1 Then
      ימים_2 = ימים_להגבלה
      ימים_2 = יומן_סינון_תאריכים(תאריך_לועזי, ימים_2, False)
      תאריך_ראשון_2 = תאריך_לשאילתה(CDate(תאריך_לועזי) - ימים_2)
   End If
   
   If סוג = 1 Then מסנן = " WHERE בר_קוד_קידומת=" & קוד_שמלה
   
   שמלות_במלאי_לפי_מידה = " (SELECT שמלות_דגמים.קטגורית_מחיר, שמלות_נתונים.בר_קוד_קידומת,  שמלות_דגמים.שם_שמלה, Sum(שמלות_נתונים.כמות) AS כמות_במלאי, שמלות_נתונים.מידה, שמלות_נתונים.מידה_טקסט" _
                        & " FROM (SELECT * FROM שמלות_דגמים " & מסנן & ") AS שמלות_דגמים INNER JOIN (SELECT * FROM שמלות_נתונים " & מסנן & ") AS שמלות_נתונים " _
                        & " ON שמלות_דגמים.בר_קוד_קידומת = שמלות_נתונים.בר_קוד_קידומת" _
                        & " WHERE שמלות_נתונים.לא_בשימוש=0 AND שמלות_נתונים.שמלה_בתיקון=False AND IIF(מיקום='רזרבה',0,-1) AND CDATE(NZ(תאריך_יציאה_מהמאגר,#23/07/2050#))" & IIf(לא_פעילים, "<", ">") & תאריך_לשאילתה_מלא(תאריך_תפוסה_אחרון) _
                        & " GROUP BY שמלות_דגמים.קטגורית_מחיר, שמלות_נתונים.בר_קוד_קידומת, שמלות_דגמים.שם_שמלה, שמלות_נתונים.מידה, שמלות_נתונים.מידה_טקסט" _
                        & ")"

   Select Case סוג
     
     ' לפי דגם
     Case 1:
       
       If חישוב_ימים = "" Then
          sql = חישוב_ימים_לפני_ואחרי(CDate(תאריך_לועזי), מהגדרות(6), תאריך_ראשון, ימים_להגבלה, , תאריך_תפוסה_אחרון)
       Else
          sql = חישוב_ימים_פתיחת_קוד(חישוב_ימים)
       End If
       sql = Replace(sql, "FROM הזמנות_פרטים", "FROM (SELECT * FROM הזמנות_פרטים WHERE הזמנות_פרטים.בר_קוד_קידומת=" & קוד_שמלה & ") AS הזמנות_פרטים")
       
       שדה_כמות = "SUM([שמלות_במלאי_לפי_מידה].[כמות_במלאי]) -NZ(SUM([סך_הכל]),0)"
       שדה_כמות = שדה_כמות & IIf(NewRecord = 0, "+ IIF([שמלות_במלאי_לפי_מידה].[מידה]='" & טק(מידה, 0) & "'," & כמות & ",0)", "")
       שדה_כמות = " IIF(" & שדה_כמות & "<0,0," & שדה_כמות & ")"
        
      If ימים_להגבלה > -1 Then
         שדה_כמות_2 = "SUM([שמלות_במלאי_לפי_מידה].[כמות_במלאי]) -NZ(SUM([סך_הכל_2]),0)"
         שדה_כמות_2 = שדה_כמות_2 & IIf(NewRecord = 0, "+ IIF([שמלות_במלאי_לפי_מידה].[מידה]='" & טק(מידה, 0) & "'," & כמות & ",0)", "")
         שדה_כמות_2 = " IIF(" & שדה_כמות_2 & "<0,0," & שדה_כמות_2 & ")"
      End If
      
      
      שמלות_מקור_שורה_לפי = " SELECT שמלות_במלאי_לפי_מידה.מידה ,שמלות_במלאי_לפי_מידה.מידה_טקסט AS [תיאור מידה]," & שדה_כמות & " AS כמות" & IIf(ימים_להגבלה > -1, "," & שדה_כמות_2 & " AS [" & ימים_להגבלה & " ימים]", "") _
         & " FROM " & שמלות_במלאי_לפי_מידה & " AS שמלות_במלאי_לפי_מידה LEFT JOIN " & sql & " AS שמלות_בתפוסה_לפי_מידה ON (שמלות_במלאי_לפי_מידה.מידה = שמלות_בתפוסה_לפי_מידה.מידה) AND (שמלות_במלאי_לפי_מידה.בר_קוד_קידומת = שמלות_בתפוסה_לפי_מידה.בר_קוד_קידומת) " _
         & " WHERE שמלות_במלאי_לפי_מידה.בר_קוד_קידומת = " & קוד_שמלה _
         & " GROUP BY שמלות_במלאי_לפי_מידה.מידה,שמלות_במלאי_לפי_מידה.מידה_טקסט" _
         & " HAVING " & IIf(ימים_להגבלה > -1, שדה_כמות_2, שדה_כמות) & " > 0" _
         & " ORDER BY שמלות_במלאי_לפי_מידה.מידה;"
     
     ' כל השמלות
     Case 2:
      
       If חישוב_ימים = "" Then
          sql = חישוב_ימים_לפני_ואחרי(CDate(תאריך_לועזי), מהגדרות(6), תאריך_ראשון, ימים_להגבלה, , תאריך_תפוסה_אחרון)
       Else
          sql = חישוב_ימים_פתיחת_קוד(חישוב_ימים)
       End If
     '  SQL = Replace(SQL, "INNER JOIN הזמנות", "INNER JOIN  (SELECT * FROM הזמנות WHERE תאריך_אירוע_לועזי<" & תאריך_אירוע + 150 & " AND תאריך_אירוע_לועזי>" & תאריך_אירוע - 150 & " ) AS הזמנות")
       
      שדה_כמות = "(SUM([שמלות_במלאי_לפי_מידה].[כמות_במלאי]) -NZ( SUM([סך_הכל]),0))"
      שדה_כמות = " IIF(" & שדה_כמות & "<0,0," & שדה_כמות & ")"
      
      If ימים_להגבלה > -1 Then
        שדה_כמות_2 = "(SUM([שמלות_במלאי_לפי_מידה].[כמות_במלאי]) - NZ(SUM([סך_הכל_2]),0))"
        שדה_כמות_2 = " IIF(" & שדה_כמות_2 & "<0,0," & שדה_כמות_2 & ")"
      End If
      
      שמלות_מקור_שורה_לפי = "(SELECT   שמלות_במלאי_לפי_מידה.מידה,שמלות_במלאי_לפי_מידה.קטגורית_מחיר, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת" & IIf(מהגדרות(26) = "כן", ", שמלות_במלאי_לפי_מידה.שם_שמלה", "") & ",  שמלות_במלאי_לפי_מידה.בר_קוד_קידומת AS קוד," & שדה_כמות & " AS כמות_בינים" & IIf(ימים_להגבלה > -1, "," & שדה_כמות_2 & " AS [כמות_בינים_2]", "") _
         & " FROM " & שמלות_במלאי_לפי_מידה & " AS שמלות_במלאי_לפי_מידה LEFT JOIN " & sql & " AS שמלות_בתפוסה_לפי_מידה ON (שמלות_במלאי_לפי_מידה.מידה = שמלות_בתפוסה_לפי_מידה.מידה) AND  (שמלות_במלאי_לפי_מידה.בר_קוד_קידומת = שמלות_בתפוסה_לפי_מידה.בר_קוד_קידומת)" _
         & " GROUP BY שמלות_במלאי_לפי_מידה.קטגורית_מחיר, שמלות_במלאי_לפי_מידה.מידה,שמלות_במלאי_לפי_מידה.בר_קוד_קידומת" & IIf(מהגדרות(26) = "כן", ", שמלות_במלאי_לפי_מידה.שם_שמלה", "") _
         & " HAVING " & IIf(ימים_להגבלה > -1, שדה_כמות_2, שדה_כמות) & " > 0 OR שמלות_במלאי_לפי_מידה.בר_קוד_קידומת IN (SELECT בר_קוד_קידומת FROM הזמנות_פרטים WHERE קוד_הזמנה= " & Nz(קוד_הזמנה, 0) & "))"
     
      שדה_כמות = "SUM([כמות_בינים])"
      שדה_כמות = שדה_כמות & IIf(קוד_שמלה > 0 And Not NewRecord, "+ IIF([בר_קוד_קידומת]=" & קוד_שמלה & "," & כמות & ",0)", "")
      שדה_כמות = " IIF(" & שדה_כמות & "<0,0," & שדה_כמות & ")  AS כמות"
    
      If ימים_להגבלה > -1 Then
         שדה_כמות_2 = "SUM([כמות_בינים_2])"
         שדה_כמות_2 = שדה_כמות_2 & IIf(קוד_שמלה > 0 And Not NewRecord, "+ IIF([בר_קוד_קידומת]=" & קוד_שמלה & "," & כמות & ",0)", "")
         שדה_כמות_2 = ", IIF(" & שדה_כמות_2 & "<0,0," & שדה_כמות_2 & ") AS [" & ימים_להגבלה & " ימים] "
      End If
      
      Dim שם_שמלה_מלא As String
      שם_שמלה_מלא = " & " & Chr(34) & " - " & Chr(34) & " & [מלאי].[קטגורית_מחיר]"
      
      שמלות_מקור_שורה_לפי = " SELECT בר_קוד_קידומת," & IIf(מהגדרות(26) = "כן", " שם_שמלה " & IIf(מהגדרות(64) = "כן", שם_שמלה_מלא, "") & " AS שם,", "") & "קוד," & שדה_כמות & IIf(ימים_להגבלה > -1, שדה_כמות_2, "") _
         & " FROM " & שמלות_מקור_שורה_לפי & " AS מלאי " _
         & " GROUP BY בר_קוד_קידומת, קוד" & IIf(מהגדרות(26) = "כן", ", שם_שמלה " & IIf(מהגדרות(64) = "כן", שם_שמלה_מלא, "") & "", "") _
         & " ORDER BY " & IIf(מהגדרות(26) = "כן", " שם_שמלה " & IIf(מהגדרות(64) = "כן", שם_שמלה_מלא, "") & "", "קוד") & ";"
    
    ' שינוי תאריך הזמנה רשימת חסרים במלאי
    Case 3:
    
    Dim חישוב(1 To 2) As String, בדיקה_מלאי_כולל_הזמנה As String, בדיקה_שמלות_בתפוסה_לפי_מידה As String
     
    בדיקה_מלאי_כולל_הזמנה = " (SELECT הזמנות_פרטים.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, SUM([כמות_במלאי]) AS כמות_במלאי_כולל, שמלות_במלאי_לפי_מידה.שם_שמלה, Sum(IIf([הזמנות_פרטים].[מחוק]=0,[כמות])) AS כמות_נדרשת" _
                         & " FROM הזמנות_פרטים LEFT JOIN " & שמלות_במלאי_לפי_מידה & " AS שמלות_במלאי_לפי_מידה ON (הזמנות_פרטים.בר_קוד_קידומת = שמלות_במלאי_לפי_מידה.בר_קוד_קידומת) AND (הזמנות_פרטים.מידה = שמלות_במלאי_לפי_מידה.מידה)" _
                         & " GROUP BY הזמנות_פרטים.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.שם_שמלה, הזמנות_פרטים.קוד_הזמנה" _
                         & " HAVING (((הזמנות_פרטים.קוד_הזמנה)=" & קוד_שמלה & ")))"
         
    בדיקה_שמלות_בתפוסה_לפי_מידה = חישוב_ימים_לפני_ואחרי(CDate(תאריך_לועזי), IIf(ימים_להגבלה = -1, מהגדרות(6), ימים_להגבלה), תאריך_ראשון, , קוד_שמלה, תאריך_תפוסה_אחרון)
    
    חישוב(1) = " NZ( SUM([כמות_במלאי_כולל]),0) -NZ(SUM([כמות_נדרשת]),0)-NZ( SUM([סך_הכל]),0) "
    חישוב(2) = " NZ( SUM([כמות_במלאי_כולל]),0)- NZ(SUM([סך_הכל]),0) "
    
    שמלות_מקור_שורה_לפי = _
       " SELECT בדיקה_מלאי_כולל_הזמנה.בר_קוד_קידומת,כמות_נדרשת, בדיקה_מלאי_כולל_הזמנה.מידה, NZ(SUM(בדיקה_מלאי_כולל_הזמנה.כמות_במלאי_כולל),0)-NZ(SUM([סך_הכל]),0) AS כמות_קיימת, בדיקה_מלאי_כולל_הזמנה.שם_שמלה, " _
       & חישוב(1) & " AS כמות_סה, " _
       & חישוב(2) & " AS כמות_בתפוסה_סה " _
       & " FROM " & בדיקה_מלאי_כולל_הזמנה & " AS בדיקה_מלאי_כולל_הזמנה LEFT JOIN " & בדיקה_שמלות_בתפוסה_לפי_מידה & " AS בדיקה_שמלות_בתפוסה_לפי_מידה ON (בדיקה_מלאי_כולל_הזמנה.מידה = בדיקה_שמלות_בתפוסה_לפי_מידה.מידה) AND (בדיקה_מלאי_כולל_הזמנה.בר_קוד_קידומת = בדיקה_שמלות_בתפוסה_לפי_מידה.בר_קוד_קידומת) " _
       & " GROUP BY  בדיקה_מלאי_כולל_הזמנה.בר_קוד_קידומת, כמות_נדרשת, בדיקה_מלאי_כולל_הזמנה.מידה,כמות_נדרשת, בדיקה_מלאי_כולל_הזמנה.שם_שמלה" _
       & " HAVING " & חישוב(1) & "<0;"
       
     ' -  יתרה לפי מידה - לא בשימוש
     Case 4:
       
       If חישוב_ימים = "" Then
          sql = חישוב_ימים_לפני_ואחרי(CDate(תאריך_לועזי), מהגדרות(6), תאריך_ראשון, ימים_להגבלה, , תאריך_תפוסה_אחרון)
       Else

Public Sub בדיקת_כמות_מקסימום_שהוזמנה()
   
   Dim sql As String, תאריך As Date, כמות_מקס As Integer, כמות As Integer, טווח_תאריכים As String, מס_שדה As Integer, יום As Integer
   
   sql = " TRANSFORM Sum(הזמנות_פרטים.כמות) AS כמות" _
       & " SELECT שמלות_דגמים.שם_שמלה, הזמנות_פרטים.מידה" _
       & " FROM (הזמנות_פרטים INNER JOIN שמלות_דגמים ON הזמנות_פרטים.בר_קוד_קידומת = שמלות_דגמים.בר_קוד_קידומת) INNER JOIN הזמנות ON הזמנות_פרטים.קוד_הזמנה = הזמנות.קוד_הזמנה" _
       & " WHERE הזמנות.תאריך_אירוע_לועזי>DATE()-200" _
       & " GROUP BY שמלות_דגמים.שם_שמלה, הזמנות_פרטים.מידה" _
       & " ORDER BY שמלות_דגמים.שם_שמלה, הזמנות_פרטים.מידה, הזמנות.תאריך_אירוע_לועזי" _
       & " PIVOT  הזמנות.תאריך_אירוע_לועזי;"
   
   ' ניסיון ליצור את הטבלה. אם היא כבר קיימת, נתעלם משגיאת היווצרות הכפולה
    On Error Resume Next
    Dim sqlCreateTable As String
    sqlCreateTable = "CREATE TABLE ספירת_מלאי (שם_שמלה TEXT(255), מידה TEXT(50), כמות INTEGER, טווח_תאריכים TEXT(255));"
    CurrentDb.Execute sqlCreateTable, dbFailOnError
    On Error GoTo 0
    
   Dim שאילתה As New ADODB.Recordset
   הפעלתשאילתה ("DELETE ספירת_מלאי.* FROM ספירת_מלאי;")
   שאילתה.Open (sql), ערכתרשומות, 2, 3
   Call פתח_שאילתה("ספירת_מלאי", "שאילתה2")
   
   Do Until שאילתה.EOF
      שאילתה2.AddNew
      שאילתה2!שם_שמלה = שאילתה!שם_שמלה
      שאילתה2!מידה = שאילתה!מידה
      כמות_מקס = 0
      יום = 2
      מס_שדה = 2
      טווח_תאריכים = ""
      
'      If "ורוד שחור רשת קצר" = שאילתה!שם_שמלה Then Stop
      Do Until שאילתה.Fields.Count - 1 = מס_שדה
         יום = 1
         תאריך = CDate(שאילתה.Fields(מס_שדה).name)
         כמות = Nz(שאילתה.Fields(מס_שדה), 0)
         Do Until יום = 4
            If שאילתה.Fields.Count - 1 < מס_שדה + יום Then Exit Do
            If DateDiff("D", תאריך, CDate(שאילתה.Fields(מס_שדה + יום).name)) < 4 Then
               כמות = כמות + Nz(שאילתה.Fields(מס_שדה + יום), 0)
            Else
               Exit Do
            End If
            יום = יום + 1
         Loop
         If כמות > כמות_מקס Then כמות_מקס = כמות: טווח_תאריכים = CDate(שאילתה.Fields(מס_שדה).name)
         מס_שדה = מס_שדה + 1
      Loop
      שאילתה2!כמות = כמות_מקס
      שאילתה2!טווח_תאריכים = טק(טווח_תאריכים, Null)
      שאילתה2.Update
      שאילתה.MoveNext
   Loop
   
End Sub

          & " WHERE בר_קוד_קידומת=" & קוד_שמלה & " AND [שמלות_בתפוסה_לפי_מידה].[תאריך_אירוע_לועזי]> CDate('" & תאריך_ראשון & "') AND [שמלות_בתפוסה_לפי_מידה].[תאריך_אירוע_לועזי]<CDate('" & תאריך_אחרון & "')" _
          & " GROUP BY שמלות_בתפוסה_לפי_מידה.בר_קוד_קידומת, שמלות_בתפוסה_לפי_מידה.מידה" _
          & " PIVOT שמלות_בתפוסה_לפי_מידה.[תאריך_אירוע_לועזי];"
          
      פתח_שאילתה (sql)
      
End Sub

Public Function פתח_טופס_תשלום(קוד_הזמנה As Long, Optional פתיחה As Boolean, Optional תאריך_פתיחה As Date, Optional שאלת_פתיחה As Boolean = -1)
   
    Dim חישוב_תשלומים As Integer, חובת_תשלום As Boolean, חישוב As Boolean
    
    חישוב_תשלומים = שמלות_תשלום_רישום(קוד_הזמנה, שאלת_פתיחה, תאריך_פתיחה)
    If חישוב_תשלומים = -1 Then חובת_תשלום = -1

    If חישוב_תשלומים <> 0 Or פתיחה Then פתח_טופס "כרטיס_הזמנה_תשלום", קוד_הזמנה & "!" & IIf(חובת_תשלום, "חובת תשלום", "") & "!!" & IIf(חישוב, "", "ללא חישוב"), פט_חלון_ראשי_בינוני
End Function

Public Function קורא_אשראי(מספרים As String)

   קורא_אשראי = (MidNum(מספרים)) & " " & Mid(MidAfter(מספרים, "="), 3, 2) & "/" & Mid(MidAfter(מספרים, "="), 1, 2)
   
End Function


Public Sub בנית_שאילתת_סיכום_צריכת_שמלות(סוג As Integer, Optional קוד_הזמנה As Variant = 0, Optional קוד_שמלה As Variant = Null, Optional מידה As Variant = Null)
    On Error GoTo שגיאה
    
    Dim תנאי_הזמנה As String
    Dim תאריך_לפני_3_חודשים As Date
    
    ' הגנה נגד Null מהטופס
    קוד_הזמנה = Nz(קוד_הזמנה, 0)
    קוד_שמלה = Nz(קוד_שמלה, 0)
    מידה = Nz(מידה, "")
    
    ' חישוב התאריך של 3 חודשים אחורה (ייעול ביצועים אדיר!)
    תאריך_לפני_3_חודשים = DateAdd("m", -3, Date)
    
    ' מוודאים שגם קוד ההזמנה וגם קוד השמלה קיימים ותקינים
    If קוד_הזמנה <> 0 And קוד_שמלה <> 0 And קוד_שמלה <> "" And קוד_שמלה <> "חסר" Then
        תנאי_הזמנה = " WHERE קוד_שמלה = " & קוד_שמלה & " AND Nz(מידה, '') = '" & מידה & "'"
    Else
        תנאי_הזמנה = ""
    End If
    
    Dim ימים_לפני As Integer, ימים_אחרי As Integer
    ימים_לפני = CInt(מהגדרות(6)) - 2
    ימים_אחרי = CInt(מהגדרות(6))
    
    Dim db As DAO.Database
    Set db = CurrentDb
    
    ' =========================================================
    ' שלב 1: בדיקה ויצירה חד-פעמית של הטבלאות (רק אם אינן קיימות)
    ' =========================================================
    On Error Resume Next
    db.Execute "CREATE TABLE tmp_DateOffsets (TargetDate DATETIME PRIMARY KEY, MinDate DATETIME, MaxDate DATETIME)", 128
    db.Execute "CREATE TABLE tmp_ActiveOrders (תאריך_אירוע TEXT(255), תאריך_אירוע_לועזי DATETIME, בר_קוד_קידומת LONG, מידה TEXT(255), כמות LONG)", 128
    db.Execute "CREATE INDEX idx_FastSearch ON tmp_ActiveOrders (בר_קוד_קידומת, מידה, תאריך_אירוע_לועזי)", 128
    On Error GoTo שגיאה
    
    ' ריקון מהיר של הנתונים הישנים (במקום למחוק את הטבלה עצמה)
    db.Execute "DELETE FROM tmp_DateOffsets", 128
    db.Execute "DELETE FROM tmp_ActiveOrders", 128
    
    ' =========================================================
    ' שלב 2: שפיכת הנתונים לטבלאות הפיזיות + סינון 3 חודשים!
    ' =========================================================
    Dim sqlInsert As String
    sqlInsert = "INSERT INTO tmp_ActiveOrders (תאריך_אירוע, תאריך_אירוע_לועזי, בר_קוד_קידומת, מידה, כמות) " _
              & "SELECT הזמנות.תאריך_אירוע, הזמנות.תאריך_אירוע_לועזי, הזמנות_פרטים.בר_קוד_קידומת, Nz(הזמנות_פרטים.מידה, ''), Sum(הזמנות_פרטים.כמות) AS כמות " _
              & "FROM הזמנות INNER JOIN הזמנות_פרטים ON הזמנות.קוד_הזמנה = הזמנות_פרטים.קוד_הזמנה " _
              & תנאי_הזמנה _
              & " GROUP BY הזמנות.תאריך_אירוע, הזמנות.תאריך_אירוע_לועזי, הזמנות_פרטים.בר_קוד_קידומת, Nz(הזמנות_פרטים.מידה, ''), הזמנות.מחוק, הזמנות_פרטים.מחוק " _
              & " HAVING (((הזמנות.מחוק) = False) And ((הזמנות_פרטים.מחוק) = False) " _
              & " And (הזמנות.תאריך_אירוע_לועזי Is Not Null) " _
              & " And (הזמנות.תאריך_אירוע_לועזי >= #" & Format(תאריך_לפני_3_חודשים, "mm\/dd\/yyyy") & "#))"
    
    db.Execute sqlInsert, 128
    
    ' שליפת תאריכי האירוע הייחודיים מהטבלה המסוננת ואכלוס חלונות הזמן
    Dim rsSource As Object, rsDest As Object
    Set rsSource = db.OpenRecordset("SELECT DISTINCT תאריך_אירוע_לועזי FROM tmp_ActiveOrders")
    Set rsDest = db.OpenRecordset("tmp_DateOffsets")
    
    Do While Not rsSource.EOF
        rsDest.AddNew
        rsDest!TargetDate = rsSource!תאריך_אירוע_לועזי
        rsDest!minDate = AddDaysSkippingWeekends(rsSource!תאריך_אירוע_לועזי, -ימים_לפני, True)
        rsDest!maxDate = AddDaysSkippingWeekends(rsSource!תאריך_אירוע_לועזי, ימים_אחרי, True)
        rsDest.Update
        rsSource.MoveNext
    Loop
    rsSource.Close: rsDest.Close
    Set rsSource = Nothing: Set rsDest = Nothing
    
    
    ' =========================================================
    ' שלב 3: הרכבת מחרוזת ה-SQL הסופית
    ' =========================================================
    Dim סיכום_מלא As String, שדה_כמות As String
    
    ' התיקון: מעבר מ-SubQuery ל-LEFT JOIN ו-GROUP BY, והעברת ה-BETWEEN לתוך פונקציית IIf כדי שאקסס לא יקרוס
    סיכום_מלא = "(SELECT O1.תאריך_אירוע, O1.תאריך_אירוע_לועזי, O1.בר_קוד_קידומת, O1.מידה AS מידה, O1.כמות AS כמות_סה, " _
              & " Sum(IIf(O2.תאריך_אירוע_לועזי BETWEEN T.MinDate AND T.MaxDate, O2.כמות, Null)) AS שדה_כמות " _
              & " FROM (tmp_ActiveOrders AS O1 INNER JOIN tmp_DateOffsets AS T ON O1.תאריך_אירוע_לועזי = T.TargetDate) " _
              & " LEFT JOIN tmp_ActiveOrders AS O2 ON (O1.בר_קוד_קידומת = O2.בר_קוד_קידומת) AND (O1.מידה = O2.מידה) " _
              & " GROUP BY O1.תאריך_אירוע, O1.תאריך_אירוע_לועזי, O1.בר_קוד_קידומת, O1.מידה, O1.כמות)"
    
    If סוג = 1 Then
        ' תיקון ה-JOIN ושילוב ההגנה מ-Null בממוצע (CInt ו-Nz) ללא מיון בעייתי בסוף
        שדה_כמות = _
          " TRANSFORM Count(שאילתה7.תאריך_אירוע_לועזי) AS Countמתוךתאריך_אירוע_לועזי" _
        & " SELECT שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי AS כמות_במלאי, Max(שאילתה7.שדה_כמות) AS מקסימום, CInt(Nz(Avg([שאילתה7].[כמות_סה]), 0)) AS ממוצע, IIF(Max(שאילתה7.שדה_כמות) > Max(שמלות_במלאי_לפי_מידה.כמות_במלאי),'בעיה',NULL) AS בעיה" _
        & " FROM " & סיכום_מלא & " AS שאילתה7 RIGHT JOIN שמלות_במלאי_לפי_מידה ON (שאילתה7.בר_קוד_קידומת = שמלות_במלאי_לפי_מידה.בר_קוד_קידומת) AND (שאילתה7.מידה = שמלות_במלאי_לפי_מידה.מידה)" _
        & " GROUP BY שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי" _
        & " ORDER BY שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה" _
        & " PIVOT שאילתה7.שדה_כמות;"
    Else
        ' שאילתת תצוגת חריגים - כאן אפשר להשאיר את ה-ORDER BY המקורי
        שדה_כמות = _
          " SELECT שאילתה7.תאריך_אירוע, שאילתה7.תאריך_אירוע_לועזי, שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי, Max(שאילתה7.שדה_כמות) AS [כמות בתפוסה]" _
        & " FROM " & סיכום_מלא & " AS שאילתה7 RIGHT JOIN שמלות_במלאי_לפי_מידה ON (שאילתה7.בר_קוד_קידומת = שמלות_במלאי_לפי_מידה.בר_קוד_קידומת) AND (שאילתה7.מידה = שמלות_במלאי_לפי_מידה.מידה)" _
        & " GROUP BY שאילתה7.תאריך_אירוע, שאילתה7.תאריך_אירוע_לועזי, שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי" _
        & " HAVING Max(שאילתה7.שדה_כמות) > Max(שמלות_במלאי_לפי_מידה.כמות_במלאי)" _
        & " ORDER BY שאילתה7.תאריך_אירוע_לועזי DESC, שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה;"
     End If
    
    CurrentDb.QueryDefs("סטטיסיטקת_צריכת_שמלות").sql = שדה_כמות
    
    Set db = Nothing ' שחרור זיכרון
Exit Sub

שגיאה:
    MsgBox "שגיאות בבנית_שאילתת_סיכום_צריכת_שמלות: " & ERR.Number & " - " & ERR.Description
    Set db = Nothing ' שחרור גם במקרה של שגיאה
End Sub

Public Sub בנית_שאילתת_סיכום_צריכת_שמלות2(סוג As Integer, Optional קוד_הזמנה As Long, Optional קוד_שמלה As Long, Optional מידה As Integer)
'On Error GoTo שגיאה
  
  Dim הזמנות_פרטים_פעיל As String
   הזמנות_פרטים_פעיל = " (SELECT הזמנות.תאריך_אירוע, הזמנות.תאריך_אירוע_לועזי, הזמנות_פרטים.בר_קוד_קידומת, הזמנות_פרטים.מידה, Sum(הזמנות_פרטים.כמות) AS כמות" _
                    & " FROM הזמנות INNER JOIN הזמנות_פרטים ON הזמנות.קוד_הזמנה = הזמנות_פרטים.קוד_הזמנה" _
                    & IIf(קוד_הזמנה <> 0, " WHERE קוד_שמלה =" & קוד_שמלה & " AND מידה = " & מידה, "") _
                    & " GROUP BY הזמנות.תאריך_אירוע, הזמנות.תאריך_אירוע_לועזי, הזמנות_פרטים.בר_קוד_קידומת, הזמנות_פרטים.מידה, הזמנות.מחוק, הזמנות_פרטים.מחוק" _
                    & " HAVING (((הזמנות.מחוק) = False) And ((הזמנות_פרטים.מחוק) = False)))"

   Dim ימים_לפני As Integer, ימים_אחרי As Integer, שדה As String, מונה As Integer, הזמנות_כל_התאריכים As String, שאילתות As String, name As String, סיכום_מלא As String

   Next
   
   הזמנות_כל_התאריכים = "(SELECT הזמנות_פרטים_פעיל.*, " & שדה & " FROM " & הזמנות_פרטים_פעיל & " AS הזמנות_פרטים_פעיל)"
   
   
   שדה = ""
   For מונה = 0 To ימים_לפני + ימים_אחרי
       שאילתות = שאילתות & "("
   Next
   
   שאילתות = שאילתות & הזמנות_כל_התאריכים & "AS  שאילתה6"
   
   For מונה = 0 - ימים_לפני To ימים_אחרי
       
       name = "  הזמנות_פרטים_פעיל_" & מונה + ימים_לפני
       שאילתות = בעמ(" ", "  LEFT JOIN הזמנות_פרטים_פעיל AS " & name & " ON (שאילתה6.[" & מונה & "] = " & name & ".תאריך_אירוע_לועזי) AND (שאילתה6.מידה = " & name & ".מידה) AND (שאילתה6.בר_קוד_קידומת = " & name & ".בר_קוד_קידומת))", שאילתות)
       שדה = בעמ("+", "NZ(הזמנות_פרטים_פעיל_" & מונה + ימים_לפני & ".כמות,0)", שדה)
   Next
   

   Dim יום_להורדה As Integer, שדה_כמות As String, מחרוזת_כמות As String: יום_להורדה = 0
   For מונה = ימים_אחרי * 2 - 2 To ימים_אחרי - 2 Step -1
       שדה_כמות = בעמ("+", "NZ([הזמנות_פרטים_פעיל_" & מונה & "].[כמות],0)", שדה_כמות)
   Next
   
   
   For מונה = ימים_אחרי + 1 To ימים_אחרי * 2 - 2
       If מחרוזת_כמות <> "" Then
          מחרוזת_כמות = "IIF(NZ([הזמנות_פרטים_פעיל_" & יום_להורדה & "].[כמות],0)-NZ([הזמנות_פרטים_פעיל_" & מונה & "].[כמות],0)+" & מחרוזת_כמות & ">0,NZ([הזמנות_פרטים_פעיל_" & יום_להורדה & "].[כמות],0)-NZ([הזמנות_פרטים_פעיל_" & מונה & "].[כמות],0)+" & מחרוזת_כמות & ",0)"
       Else
         מחרוזת_כמות = "IIF(NZ([הזמנות_פרטים_פעיל_" & יום_להורדה & "].[כמות],0)-NZ([הזמנות_פרטים_פעיל_" & מונה & "].[כמות],0)>0,NZ([הזמנות_פרטים_פעיל_" & יום_להורדה & "].[כמות],0)-NZ([הזמנות_פרטים_פעיל_" & מונה & "].[כמות],0),0)"
       End If
       יום_להורדה = יום_להורדה + 1
   Next
   
   שדה_כמות = בעמ("+", מחרוזת_כמות, שדה_כמות)
   
   סיכום_מלא = "(SELECT שאילתה6.*," & שדה & " AS כמות_סה, " & שדה_כמות & " AS שדה_כמות FROM " & שאילתות & ")"
  
   If סוג = 1 Then
     שדה_כמות = _
       " TRANSFORM Count(שאילתה7.תאריך_אירוע_לועזי) AS Countמתוךתאריך_אירוע_לועזי" _
     & " SELECT  שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי AS כמות_במלאי, Max(שאילתה7.שדה_כמות) AS מקסימום, CInt(Avg([שאילתה7].[כמות_סה])) AS ממוצע, IIF(Max(שאילתה7.כמות_סה) > Max(שמלות_במלאי_לפי_מידה.כמות_במלאי),'בעיה',NULL) AS בעיה" _

     & " GROUP BY שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שמלות_במלאי_לפי_מידה.כמות_במלאי" _
     & " ORDER BY שמלות_במלאי_לפי_מידה.שם_שמלה, שמלות_במלאי_לפי_מידה.בר_קוד_קידומת, שמלות_במלאי_לפי_מידה.מידה, שאילתה7.שדה_כמות DESC" _
     & " PIVOT שאילתה7.שדה_כמות;"
   Else

Private Function GetValue(jsonText As String, key As String) As String
    Dim p1 As Long, p2 As Long
    Dim searchKey As String

    searchKey = """" & key & """"

    p1 = InStr(jsonText, searchKey)
    If p1 = 0 Then Exit Function

    p1 = InStr(p1, jsonText, ":")
    p1 = InStr(p1, jsonText, """") + 1
    p2 = InStr(p1, jsonText, """")

    GetValue = Mid(jsonText, p1, p2 - p1)
End Function

Public Function GetLinkedPath()
    ' פונקציה זו מיועדת להיקרא על ידי ה-HTA
    ' היא מחזירה את הנתיב האמיתי של טבלת לוח_שנה
    
    Dim db As DAO.Database
    Dim tDef As DAO.TableDef
    Dim strConnect As String
    Dim strPath As String
    
    On Error GoTo ErrorHandler
    
    Set db = CurrentDb
    
    ' ננסה לגשת לטבלה הספציפית
    Set tDef = db.TableDefs("לוח_שנה")
    
    strConnect = tDef.Connect
    
    ' אם המחרוזת ריקה, סימן שהטבלה מקומית
    If strConnect = "" Then
        GetLinkedPath = CurrentProject.Fullname
    Else
        ' המחרוזת נראית כך: ;DATABASE=C:\Folder\Data.accdb
        ' אנחנו צריכים לחלץ את הנתיב אחרי המילה DATABASE=
        Dim pStart As Integer
        pStart = InStr(1, strConnect, "DATABASE=")
        
        If pStart > 0 Then
            strPath = Mid(strConnect, pStart + 9) ' 9 זה אורך המילה DATABASE=
            
            ' ניקוי תווים מיותרים בסוף אם יש (כמו ; נוסף)
            If InStr(strPath, ";") > 0 Then
                strPath = Left(strPath, InStr(strPath, ";") - 1)
            End If
            
            GetLinkedPath = strPath
        Else
            ' מקרה נדיר של חיבור מוזר
            GetLinkedPath = CurrentProject.Fullname
        End If
    End If
    
    Exit Function

ErrorHandler:
    ' במקרה של שגיאה (למשל הטבלה לא קיימת), נחזיר את הנתיב הנוכחי
    GetLinkedPath = CurrentProject.Fullname
End Function

        daysBefore:=CLng(מהגדרות(6)), _
        daysAfter:=CLng(מהגדרות(6)), _
        skipWeekends:=-1, _
        isWeekdayEvent:=isMultiDay, _
        newFromDate:=varFrom, _
        newToDate:=varTo _
    )
    
    ' אם הפונקציה לא החזירה שאילתה (בגלל תאריכים שגויים), נעצור מיד.
    If sqlCheck = "" Then Exit Function
    
    On Error GoTo Error_Handler
    
    On Error Resume Next
    db.QueryDefs.Delete "DEBUG_Shortages"
    db.CreateQueryDef "DEBUG_Shortages", sqlCheck
    db.QueryDefs.Refresh
    On Error GoTo Error_Handler
    
    On Error Resume Next
    DoCmd.DeleteObject acTable, "tbl_TempShortages"
    On Error GoTo Error_Handler
    
    sqlMakeTable = "SELECT * INTO tbl_TempShortages FROM (" & sqlCheck & ")"
    db.Execute sqlMakeTable, 128 ' dbFailOnError
    
    If DCount("*", "tbl_TempShortages") = 0 Then
        שנה_תאריך = True
    Else
        תשובה_ראשית = ללא
        params = קוד_הזמנה & "!" & תאריך_חדש & "!חלק מהפריטים תפוסים / חסרים במלאי!שינוי תאריך!" & Nz(תאריך_תפוסה_אחרון, "")
        פתח_טופס "כרטיס_הזמנה_שינוי_תאריך_פירוט", params, פט_חלון_ראשי, True
        If תשובה_ראשית = אישור Then שנה_תאריך = True
    End If
    
    If שנה_תאריך Then
        If שנה Then
            הפעלתשאילתה "UPDATE הזמנות SET תאריך_אירוע_לועזי =" & תאריך_לשאילתה_מלא(תאריך_חדש) & _
                         ", תאריך_אירוע = '" & ללא_גרש(המרה_לעברי(תאריך_חדש)) & _
                         "' WHERE קוד_הזמנה=" & קוד_הזמנה & ";"
        End If
        If שם_טופס <> "" Then סגור_טופס שם_טופס
        שינוי_תאריך_הזמנה = True
    End If

Clean_Exit:
    On Error Resume Next
    db.QueryDefs.Delete "tmpRescheduleAgg_Fixed"
    Exit Function

Error_Handler:
    MsgBox "אירעה שגיאה. ייתכן ששם עמודה לא תקין. פירוט שגיאה: " & ERR.Number & " - " & ERR.Description, vbCritical
    Resume Clean_Exit
End Function
